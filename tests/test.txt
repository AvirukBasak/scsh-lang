module main

proc main start
    print("Hello world!", lf)

    x = input(int)

    if x == 1 then
        print("entered 1", lf)
    elif x == 2 then
        print("entered 2", lf)
    else
        print(f"entered {x}", lf)
    end

    while x >= 0 do
        print(f"x = {x}")
        x = x -1
    end

    x = input("prompt: ", str)
    print(f"entered {x}", lf)
end
