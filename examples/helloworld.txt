module main

proc test start
    var times = $1
    for x in times do
        var name = input(f"Enter name {x}: ", str)
        var bytes = print(f"Hello {name}!",
            "It's a beautiful world!", lf)
        print(f"Printed: {bytes} Bytes", lf, lf)
    end
end

proc main start
    var ret = test("Times", [1, 2, 3, 4, 5])
    print(ret, lf)
end
