proc LinkedList()
    var l = {}
    l.head = null
    l.append = LinkedList_append
    l.popf = LinkedList_popf
    l.show = LinkedList_show
    return l
end

proc LinkedList_append(val)
    var tmp = this.head
    this.head = { val: val, next: tmp }
end

proc LinkedList_popf()
    var tmp = this.head
    this.head = this.head.next
    tmp = null
end

proc LinkedList_show()

    while ptr do
        io:print(ptr.val, "")

end
