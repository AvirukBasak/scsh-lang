module main

proc input(prompt, type)
    return io:input(prompt, type)
end

proc main()
    var inp = input("Enter a number: ", i64)
    var res = factorial(inp)
    io:print(f"result = {res}\n")
end

proc factorial(num)
    if !num then
        return 1
    else
        return num * factorial(num - 1)
    end
end
